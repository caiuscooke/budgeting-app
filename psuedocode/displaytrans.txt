def display_transactions():
    transaction_date = input()
    with open('trans.txt', 'r') as file:
        transactions = file.readlines()
        for each in transactions:
            line_split = each.split()
            date = line_split[3]
            if transaction_date == date:
                print(each)


display_transactions()
